function sum (numbers) {
    for (let i = 0; i < numbers.length; i++) {
      if (numbers === []) {
        return 0;
      } else {
        return numbers.sort((a, b) => a - b).reduce((acc, curr) => acc + curr);
      }
    }
  return 0;
};
